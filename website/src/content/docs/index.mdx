  <div class="gluegun-lab">
    <div class="gluegun-lab__intro">
      <div>
        <p class="gluegun-lab__signal">Start with one collected response.</p>
        <h2 id="gluegun-lab-title">Typed HTTP without hiding Gun.</h2>
        <p>
          Gluegun keeps the BEAM-native Gun model visible: open a connection, wait for the negotiated protocol, send the request, then collect a typed response or drop into streams when the body cannot fit in memory.
        </p>
      </div>
    </div>
    <div class="gluegun-lab__proof">
      <pre class="gluegun-lab__code" aria-label="Gleam example using Gluegun"><code>{`let assert Ok(conn) =
  connection.options()
  |> connection.open(host: "example.com", port: 443)

let assert Ok(_) =
  connection.await_up(conn, connection.Milliseconds(5000))

let assert Ok(res) =
  client.new(request.Get, "/")
  |> client.with_timeout(timeout: connection.Milliseconds(5000))
  |> client.send(connection: conn)

io.println("status: " <> int.to_string(response.status(res)))
let assert Ok(Nil) = connection.close(conn)`}</code></pre>
      <p class="gluegun-lab__caption">
        `client.send` collects status, headers, body, trailers, and informational `1xx` responses into one typed value. Use the lower-level request and message APIs when the response must stream.
      </p>
    </div>
