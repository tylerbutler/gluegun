      </ul>
    </div>
    <pre class="gluegun-lab__code" aria-label="Gleam example using Gluegun"><code>{`let assert Ok(conn) =
  connection.options()
  |> connection.open(host: "example.com", port: 443)

let assert Ok(protocol) =
  connection.await_up(conn, connection.Milliseconds(5000))

client.new(request.Get, "/")
|> client.with_timeout(timeout: connection.Milliseconds(5000))
|> client.send(connection: conn)`}</code></pre>
